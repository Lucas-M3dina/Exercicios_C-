algoritmo "Media de notas"

VAR
   STRING NomePessoa, media;
   FLOAT NOTA1, NOTA2, NOTA3, MEDIA;

INICIO

      ESCREVA ("Qual é o nome do aluno??")
      LEIA (NomePessoa)

      ESCREVA ("Digite a primeira nota")
      LEIA NOTA1

      ESCREVA ("Digite a segunda nota")
      LEIA NOTA2

      ESCREVA ("Digite a terceira nota")
      LEIA NOTA3

      media = (NOTA1 + NOTA2 + NOTA3) / 3;

      condicao = SE A MEDIA FOR >= 7 GUARDE APROVADO SE NÃO GUARDE REPROVADO

      ESCREVA ("O ALUNO" + NomePessoa + "TEVE A MEDIA" + media + "E FOI" + condicao)

FIM DO ALGORITMO
